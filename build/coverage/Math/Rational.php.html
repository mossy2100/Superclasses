577    /**
578     * Convert a fraction to its canonical form.
579     *
580     * 1. Use canonical form for 0 (0/1).
581     * 2. Reduce fraction to its simplest form.
582     * 3. Ensure denominator is positive.
583     *
584     * @param int $num The numerator.
585     * @param int $den The denominator.
586     * @return int[] The simplified numerator and denominator.
587     */
588    private static function simplify(int $num, int $den): array
589    {
590        // Check for 0.
591        if ($num === 0) {
592            return [0, 1];
593        }
594
595        // Check for 1.
596        if ($num === $den) {
597            return [1, 1];
598        }
599
600        // Check for -1.
601        if ($num === -$den) {
602            return [-1, 1];
603        }
604
605        // Handle PHP_INT_MIN specially because otherwise the call to abs() would overflow.
606        // We may be able to solve this by dividing by 2. PHP_INT_MIN equals -2^n (where n is the bit width minus 1),
607        // therefore 2 is the only common factor to consider.
608        if (($num === PHP_INT_MIN && $den % 2 === 0) || ($den === PHP_INT_MIN && $num % 2 === 0)) {
609            $num = intdiv($num, 2);
610            $den = intdiv($den, 2);
611        }
612
613        // Handle numerator of PHP_INT_MIN.
614        if ($num === PHP_INT_MIN) {
615            // We can return the values as-is if the denominator is positive. We know the fraction will already be
616            // reduced to its simplest form due to the previous block. If we're here, the denominator is odd.
617            if ($den > 0) {
618                return [$num, $den];
619            }
620
621            // Since the canonical form has a positive denominator, we can't return this as-is, and have to throw an
622            // exception.
623            throw new RangeException("Cannot simplify a rational with a numerator of PHP_INT_MIN and a negative denominator.");
624        }
625
626        // Handle denominator of PHP_INT_MIN.
627        if ($den === PHP_INT_MIN) {
628            // Since the canonical form has a positive denominator, we can't return this as-is, and have to throw an
629            // exception.
630            throw new RangeException("Cannot simplify a rational with a denominator of PHP_INT_MIN.");
631        }
632
633        // Calculate the GCD.
634        $gcd = Numbers::gcd($num, $den);
635
636        // Reduce the fraction if necessary.
637        if ($gcd > 1) {
638            $num = intdiv($num, $gcd);
639            $den = intdiv($den, $gcd);
640        }
641
642        // Ensure the denominator is positive.
643        if ($den < 0) {
644            return [-$num, -$den];
645        }
646
647        // Return the simplified fraction.
648        return [$num, $den];
649    }
