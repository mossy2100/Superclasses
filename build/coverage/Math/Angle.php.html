491     * Normalize a scalar angle value into a specified half-open interval.
492     *
493     * If $signed is false (default), the range is [0, $units_per_turn).
494     * If $signed is true, the range is [-$units_per_turn/2, $units_per_turn/2).
495     *
496     * @param float $value The value to wrap.
497     * @param float $units_per_turn Units per full turn (e.g., TAU for radians, 360 for degrees, 400 for gradians).
498     * @param bool $signed Whether to return a signed range instead of the default positive range.
499     * @return float The wrapped value.
500     */
501    private static function _wrapScalar(float $value, float $units_per_turn, bool $signed = false): float
502    {
503        // Reduce using fmod to avoid large magnitudes.
504        $r = fmod($value, $units_per_turn);
505
506        // Get the range bounds.
507        $half = $units_per_turn / 2.0;

509        $max  = $signed ? $half : $units_per_turn;
510
511        // Adjust into the half-open interval [min, max).
512        if ($r < $min) {
513            $r += $units_per_turn;
514        }
515        elseif ($r >= $max) {
516            $r -= $units_per_turn;
517        }
518
519        // Canonicalize -0.0 to 0.0.
520        if ($r == 0.0) {
521            $r = 0.0;
522        }
523
524        return $r;
525    }

673    /**
674     * Checks that the input string, which is meant to indicate an angle, is valid.
675     *
676     * Different units (deg, rad, grad, turn) are supported, as used in CSS.
677     * There can be spaces between the number and the unit.
678     * @see https://developer.mozilla.org/en-US/docs/Web/CSS/angle
679     *
680     * The format with degrees, minutes, and seconds, as produced by the toDMSString() method, is
681     * also supported.
682     * There cannot be any space between a number and its unit, but it's ok to have a single space
683     * between two parts.
684     *
685     * If valid, the angle is returned; otherwise, an exception is thrown.
686     *
687     * @param string $value The string to parse.
688     * @return self A new angle equivalent to the provided string.
689     * @throws DomainException If the string does not represent a valid angle.
690     */
691    public static function fromString(string $value): self
692    {
693        // Prepare an error message with the original value.
694        $err_msg = "The provided string '$value' does not represent a valid angle.";
695
696        // Reject empty input.
697        $value = trim($value);
698        if ($value === '') {
699            throw new DomainException($err_msg);
700        }
701
702        // Check for the DMS pattern as returned by toDMSString().
703        // That means optional minus, then optional degrees, minutes, seconds.
704        $num     = '(?:\d+(?:\.\d+)?|\.\d+)';
705        $pattern = "/^(?:(?P<sign>[-+]?)\s*)?"
706                   . "(?:(?P<deg>{$num})°\s*)?"
707                   . "(?:(?P<min>{$num})[′']\s*)?"
708                   . "(?:(?P<sec>{$num})[″\"])?$/u";
709        if (preg_match($pattern, $value, $matches)) {

711            if (empty($matches['deg']) && empty($matches['min']) && empty($matches['sec'])) {
712                throw new DomainException($err_msg);
713            }
714
715            // Get the sign.
716            $sign = isset($matches['sign']) && $matches['sign'] === '-' ? -1 : 1;
717
718            // Extract the parts.
719            $d = isset($matches['deg']) ? $sign * (float)$matches['deg'] : 0.0;
720            $m = isset($matches['min']) ? $sign * (float)$matches['min'] : 0.0;
721            $s = isset($matches['sec']) ? $sign * (float)$matches['sec'] : 0.0;
722
723            // Convert to angle.
724            return self::fromDMS($d, $m, $s);
725        }
726
727        // Check for units.
728        if (preg_match("/^(-?$num)\s*(rad|deg|grad|turn)$/i", $value, $m)) {
729            $num = (float)$m[1];
730            return match (strtolower($m[2])) {

732                'deg'  => self::fromDegrees($num),
733                'grad' => self::fromGradians($num),
734                'turn' => self::fromTurns($num),
